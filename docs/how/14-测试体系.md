## 14.1 设计动机

**测试不是事后补丁**,是重构的**安全保障**。本项目测试覆盖三大维度:

1. **路由分类**:Router 把用户输入正确转 `AgentAction`
2. **引擎分支**:8 种 Done reason 都能正确产出
3. **Tool 安全**:路径越界、白名单、AGENT.md guard 全部拒绝

**框架选择**:

- **pytest + pytest-asyncio**(`asyncio_mode = "auto"`):不需要显式 `@pytest.mark.asyncio`
- **pydantic**:`AgentAction` / `Agent` / `SkillDirective` 都是 BaseModel,直接 `.model_validate(...)` 校验测试数据
- **tmp_path fixture**:pytest 内置临时目录,适合测文件落盘

```python
# tests/test_routing.py
import pytest
from writer.routing import RuleBasedIntentRouter, AgentAction


@pytest.fixture
def router() -> RuleBasedIntentRouter:
    return RuleBasedIntentRouter()


class TestRuleBasedRouter:
    def test_init_command(self, router):
        action = router.route("/init 我的小说", "S0")
        assert action.action_type == "run_command"
        assert action.command == "/init"

    def test_wordcount_command(self, router):
        action = router.route("/字数统计 outline/", "S1")
        assert action.action_type == "call_tool"
        assert action.tool_name == "wordcount"
        assert action.arguments == {"path": "outline/"}

    def test_foreshadow_with_id(self, router):
        action = router.route("查一下伏笔 F003", "S2")
        assert action.tool_name == "foreshadow_search"
        assert action.arguments["id"] == "F003"
        assert "F003" in action.arguments["keyword"]

    def test_foreshadow_without_id(self, router):
        action = router.route("查一下伏笔 玉佩来历", "S2")
        assert action.tool_name == "foreshadow_search"
        assert action.arguments == {"keyword": "查一下伏笔 玉佩来历"}

    def test_create_command(self, router):
        action = router.route("/创作 第3章", "S3")
        assert action.action_type == "start_workflow"
        assert action.workflow == "write_chapter"

    def test_natural_language_fallback(self, router):
        action = router.route("你好", "S1")
        assert action.action_type == "answer_directly"
        assert "我可以处理" in action.answer

    def test_framework_keyword_detected(self, router):
        assert RuleBasedIntentRouter.looks_like_command("退出") is True
        assert RuleBasedIntentRouter.looks_like_command("/大纲") is True
        assert RuleBasedIntentRouter.looks_like_command("随机文本") is False
```

### 模式 2:Engine Done 分支测试

```python
# tests/test_engine.py
import pytest
from writer.runner import Engine, Done, ActionEvent, ToolCall, ToolResult
from writer.runner.context import RunnerContext


@pytest.fixture
def deps_with_rule_router():
    """装配一个不带 LLM 的 deps,纯 rule-based。"""
    from writer.runner.deps import production_deps
    return production_deps(settings=...)


@pytest.mark.asyncio
async def test_answered_branch(deps_with_rule_router, tmp_path):
    """test_router_classifies_write_command → /大纲 走 directive → answered"""
    ctx = RunnerContext(
        user_input="/大纲 穿越到唐朝",
        project_root=tmp_path,
        project_state="S1",
    )
    events = []
    engine = Engine(deps=deps_with_rule_router)
    async for event in engine.run(ctx):
        events.append(event)

    assert any(isinstance(e, Done) and e.reason == "answered" for e in events)
    assert any(isinstance(e, ActionEvent) for e in events)


@pytest.mark.asyncio
async def test_command_pending_branch(deps_with_rule_router, tmp_path):
    """未注册的 /xxx 命令 → command_pending"""
    ctx = RunnerContext(
        user_input="/notreal command",
        project_root=tmp_path,
        project_state="S1",
    )
    events = []
    engine = Engine(deps=deps_with_rule_router)
    async for event in engine.run(ctx):
        events.append(event)
    last_done = next(e for e in events if isinstance(e, Done))
    assert last_done.reason == "command_pending"


@pytest.mark.asyncio
async def test_tool_completed_branch(deps_with_rule_router, tmp_path):
    """rule-only 路径:wordcount → tool_completed"""
    (tmp_path / "AGENT.md").write_text("# test\n")
    ctx = RunnerContext(
        user_input="/字数统计",
        project_root=tmp_path,
        project_state="S1",
    )
    events = []
    engine = Engine(deps=deps_with_rule_router)
    async for event in engine.run(ctx):
        events.append(event)
    assert any(isinstance(e, ToolCall) and e.name == "wordcount" for e in events)
    assert any(isinstance(e, ToolResult) and e.name == "wordcount" for e in events)
    last_done = next(e for e in events if isinstance(e, Done))
    assert last_done.reason == "tool_completed"
```

> 新代码应使用 `Engine(deps=...).run(ctx)`。`engine.loop.run_runner(ctx, deps)` 是 compat shim —— 现有测试不动。

### 模式 3:LangChain Fake ChatModel(per 备忘 09)

```python
# tests/test_llm.py
from typing import Any
from langchain_core.language_models import BaseChatModel
from langchain_core.messages import AIMessage, BaseMessage
from langchain_core.outputs import ChatGeneration, ChatResult
from pydantic import PrivateAttr


class FakeChatModel(BaseChatModel):
    """测试用 fake model:按顺序返回预定义的 AIMessage。"""
    _script: list[AIMessage] = PrivateAttr(default_factory=list)
    _calls: list[list[BaseMessage]] = PrivateAttr(default_factory=list)

    @property
    def _llm_type(self) -> str:
        return "fake"

    def bind_tools(self, tools, **kwargs) -> "FakeChatModel":
        # 直接返回 self;循环用同样的 _script
        return self

    def _generate(self, messages, stop=None, run_manager=None, **kwargs) -> ChatResult:
        self._calls.append(messages)
        if not self._script:
            return ChatResult(generations=[ChatGeneration(message=AIMessage(content=""))])
        return ChatResult(generations=[ChatGeneration(message=self._script.pop(0))])

    async def _agenerate(self, messages, stop=None, run_manager=None, **kwargs) -> ChatResult:
        return self._generate(messages, stop, run_manager, **kwargs)


def make_loop_with_fake(settings, registry, runtime, script: list[AIMessage]):
    from writer.llm.agent import ReActAgent
    fake = FakeChatModel(_script=list(script))
    return ReActAgent(settings=settings, registry=registry, runtime=runtime, llm=fake)
```

**关键约束**:
- `_llm_type` 必须实现(否则 LC 抛错)
- `_script` / `_calls` 用 `PrivateAttr`(pydantic 不允许普通属性)
- `bind_tools` 返回 self(桩,避免 LC 重新构造 model)
- `_generate` 返回 `ChatResult(generations=[ChatGeneration(message=...)])` 不是 `AIMessage`

### 模式 6:错误边界测试

```python
@pytest.mark.asyncio
async def test_engine_catches_tool_error(deps_with_rule_router, tmp_path):
    """Tool 抛 ToolError 时,引擎捕获并产 ErrorEvent + Done(aborted)。"""
    from writer.tools.errors import ToolError

    # 构造一个必抛错的 registry stub
    class FailingTool:
        name = "fail"
        description = "测试工具"
        def run(self, runtime, **kwargs):
            raise ToolError("故意失败")

    failing_registry = type("R", (), {"invoke": lambda *a, **kw: (_ for _ in ()).throw(ToolError("x"))})()

    deps = deps_with_rule_router
    deps.tool_registry = failing_registry

    ctx = RunnerContext(
        user_input="/字数统计",
        project_root=tmp_path,
        project_state="S1",
    )
    events = []
    engine = Engine(deps=deps)
    async for event in engine.run(ctx):
        events.append(event)

    assert any(isinstance(e, ErrorEvent) for e in events)
    last_done = next(e for e in events if isinstance(e, Done))
    assert last_done.reason == "aborted"
```
